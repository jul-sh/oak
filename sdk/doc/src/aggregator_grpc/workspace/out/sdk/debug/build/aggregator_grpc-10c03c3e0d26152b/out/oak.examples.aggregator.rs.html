#[derive(Clone, PartialEq, ::prost::Message)]
pub struct SerializedSparseVector {
    #[prost(uint32, repeated, tag="1")]
    pub indices: ::prost::alloc::vec::Vec<u32>,
    #[prost(float, repeated, tag="2")]
    pub values: ::prost::alloc::vec::Vec<f32>,
}
/// Message sent by clients.
#[derive(Clone, PartialEq, ::prost::Message)]
pub struct Sample {
    /// Data with different Bucket IDs is not mixed and aggregated in different instances of the
    /// Aggregator, and each has its own minimal data sample threshold.
    #[prost(string, tag="1")]
    pub bucket: ::prost::alloc::string::String,
    #[prost(message, optional, tag="2")]
    pub data: ::core::option::Option<SerializedSparseVector>,
}
